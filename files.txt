@csrf_exempt
def save_project(request, project_id=None):
    pid = project_id
    print(f"Data: {request.body}")
    if request.method != 'POST':
        return HttpResponse("false", content_type="text/plain", status=405)

    if not pid:  # first save, no ID yet
        proj = Project.objects.create(
            title="Untitled-1",
            owner=None,
            data=request.body
        )
        return JsonResponse({
    "status": "ok",
    "autosave-interval": 120,
    "content-title": base64.b64encode(proj.title.encode("utf-8")).decode("utf-8"),
    "content-name": str(proj.id)
})

    # Normal update
    proj = get_object_or_404(Project, pk=pid)
    proj.data = request.body
    proj.save()
    return JsonResponse({
    "status": "ok",
    "autosave-interval": 120,
    "content-title": base64.b64encode(proj.title.encode("utf-8")).decode("utf-8"),
    "content-name": str(proj.id)
})

def set_project_thumbnail(request, project_id):
    if not request.body:
        return HttpResponseBadRequest("No image data")

    project = get_object_or_404(Project, pk=project_id)
    filename = f"thumb_{project_id}.png"
    project.thumbnail.save(filename, request.FILES.get('file') or request, save=True)

    return HttpResponse("ok")

@csrf_exempt
def asset_set(request, asset_id):
    print(f"Raw asset id: {asset_id}")
    assetExt = asset_id.split('.')[-1]
    asset_id = clean_asset_id(asset_id) + "." + assetExt
    if request.method != 'POST':
        return HttpResponse(status=405)

    cleaned_id = SAFE_ASSET_RE.sub('', asset_id)
    assets_dir = os.path.join(settings.MEDIA_ROOT, 'assets')
    os.makedirs(assets_dir, exist_ok=True)

    file_path = os.path.join(assets_dir, cleaned_id)
    with open(file_path, 'wb') as f:
        f.write(request.body)

    return HttpResponse("true", content_type="text/plain")

@csrf_exempt
def asset_get(request, asset_id):
    assetExt = asset_id.split('.')[-1]
    asset_id = clean_asset_id(asset_id) + "." + assetExt
    print(asset_id)
    file_path = os.path.join(settings.MEDIA_ROOT, 'assets', asset_id)
    if not os.path.exists(file_path):
        return HttpResponseNotFound("asset not found")
    
    print("Getting asset")

    ext = asset_id.lower().split('.')[-1]
    content_type = {
        'svg': 'image/svg+xml',
        'png': 'image/png',
        'wav': 'audio/wav',
        'mp3': 'audio/mpeg'
    }.get(ext, 'application/octet-stream')

    with open(file_path, 'rb') as f:
        return HttpResponse(f.read())

from django.db import models
from django.contrib.auth.models import User

class Project(models.Model):
    title = models.CharField(max_length=255, default="Untitled")
    owner = models.ForeignKey(User, on_delete=models.CASCADE, null=True, blank=True)
    data = models.BinaryField(blank=True, null=True)  # store .sb2 zip or JSON
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    thumbnail = models.ImageField(upload_to='thumbnails/', null=True, blank=True)
    #data = models.JSONField(default=dict, blank=True)
    modified = models.DateTimeField(auto_now=True)  # <-- this is the magic


    def __str__(self):
        return self.title or f"Project {self.id}"
